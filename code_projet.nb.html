p<-4

AIC<-matrix(0,p,q)
BIC<-matrix(0,p,q)


for(i in 1:p){
  for(j in 1:q){
    model<-arima(nvidia_ts_dif,order = c((i-1),0,(j-1)))
    if (Box.test(model$residuals)$p.value>0.05){
      AIC[i,j]<-AIC(model)
      BIC[i,j]<-BIC(model)
    } else {
      AIC[i,j]<-1
      BIC[i,j]<-1
    }
      
    
  }
}
